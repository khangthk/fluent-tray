98 bool string2wstring(const std::string& str, std::wstring& wstr) {
99 if(str.empty()) {
100 wstr.clear() ;
101 return true ;
102 }
103
104 auto needed_size = MultiByteToWideChar(
105 CP_UTF8, 0,
106 str.c_str(), static_cast<int>(str.length()),
107 NULL, 0) ;
108 if(needed_size <= 0) {
109 return false ;
110 }
111
112 wstr.resize(needed_size) ;
113 if(MultiByteToWideChar(
114 CP_UTF8, 0,
115 str.c_str(), static_cast<int>(str.length()),
116 &wstr[0], static_cast<int>(wstr.length())) <= 0) {
117 return false;
118 }
119 return true ;
120 }
121
129 bool wstring2string(const std::wstring& wstr, std::string& str) {
130 if(wstr.empty()) {
131 str.clear() ;
132 return true ;
133 }
134
135 auto needed_size = WideCharToMultiByte(
136 CP_UTF8, 0,
137 wstr.c_str(), static_cast<int>(wstr.length()),
138 NULL, 0,
139 NULL, NULL) ;
140 if(needed_size <= 0) {
141 return false ;
142 }
143
144 str.resize(needed_size) ;
145 if(WideCharToMultiByte(
146 CP_UTF8, 0,
147 wstr.c_str(), static_cast<int>(wstr.length()),
148 &str[0], static_cast<int>(str.size()),
149 NULL, NULL) <= 0) {
150 return false ;
151 }
152 return true ;
153 }

160 inline constexpr std::size_t bit2mask(std::size_t bits) noexcept {
161 return (static_cast<std::size_t>(1) << bits) - 1 ;
162 }
163
168 template <typename Type>
169 inline constexpr int type2bit() noexcept {
170 return static_cast<int>(sizeof(Type) * CHAR_BIT) ;
171 }
172
181 template <typename InType, typename OutType>
182 inline void split_bits(InType input, OutType& upper, OutType& lower) noexcept {
183 static constexpr auto bits = type2bit<OutType>() ;
184 static const auto lower_mask = util::bit2mask(bits) ;
185
186 upper = static_cast<OutType>(reinterpret_cast<std::size_t>(input) >> bits) ;
187 lower = static_cast<OutType>(reinterpret_cast<std::size_t>(input) & lower_mask) ;
188 }
189
197 template <typename InType, typename OutType>
198 inline void concatenate_bits(InType upper, InType lower, OutType& out) noexcept {
199 static constexpr auto bits = type2bit<InType>() ;
200 static const auto lower_mask = util::bit2mask(bits) ;
201
202 auto out_upper = static_cast<std::size_t>(upper) << bits ;
203 auto out_lower = static_cast<std::size_t>(lower) & lower_mask ;
204 out = reinterpret_cast<OutType>(out_upper | out_lower) ;
205 }

536 bool draw_menu(LPDRAWITEMSTRUCT info, HFONT font) const {
537 if(SetTextColor(info->hDC, text_color_) == CLR_INVALID) {
538 return false ;
539 }
540
541 if(SetBkColor(info->hDC, back_color_) == CLR_INVALID) {
542 return false ;
543 }
544
545 if(font) {
546 if(!SelectObject(info->hDC, font)) {
547 return false ;
548 }
549 }
550
551 LONG checkmark_size, label_height, icon_size, margin ;
552 if(!calculate_layouts(
553 info->hDC,
554 checkmark_size, label_height, icon_size, margin)) {
555 return false ;
556 }
557
558 auto& rect = info->rcItem ;
559
560 auto y_center = rect.top + (rect.bottom - rect.top) / 2 ;
561 auto x = rect.left + margin ;
562
563 if(toggleable_ && checked_) {
564 if(!TextOutW(
565 info->hDC, x, y_center - label_height / 2, checkmark_.c_str(),
566 static_cast<int>(checkmark_.length()))) {
567 return false ;
568 }
569 }
570 x += checkmark_size + margin ;
571
572 if(hicon_) {
573 if(!DrawIconEx(
574 info->hDC, x, y_center - SM_CYICON / 2, hicon_,
575 icon_size, icon_size, 0, NULL, DI_NORMAL)) {
576 return false ;
577 }
578 }
579 x += icon_size + margin ;
580
581 if(!TextOutW(
582 info->hDC, x, y_center - label_height / 2, label_.c_str(),
583 static_cast<int>(label_.length()))) {
584 return false ;
585 }
586
587 if(under_line_) {
588 auto original_obj = SelectObject(info->hDC, GetStockObject(DC_PEN)) ;
589 if(SetDCPenColor(info->hDC, border_color_) == CLR_INVALID) {
590 return false ;
591 }
592
593 auto lx = info->rcItem.left ;
594 auto ly = info->rcItem.bottom - 1 ;
595 auto rx = info->rcItem.right ;
596 auto ry = ly + 1 ;
597 if(!Rectangle(info->hDC, lx, ly, rx, ry)) {
598 return false ;
599 }
600
601 if(!SelectObject(info->hDC, original_obj)) {
602 return false ;
603 }
604 }
605
606 return true ;
607 }

622 bool calculate_required_dims(HFONT font, SIZE& size) const {
623 auto hdc = GetDC(hwnd_) ;
624 if(font) {
625 if(!SelectObject(hdc, font)) {
626 return false ;
627 }
628 }
629
630 if(!GetTextExtentPoint32W(
631 hdc, label_.c_str(),
632 static_cast<int>(label_.length()), &size)) {
633 return false ;
634 }
635
636 LONG checkmark_size, label_height, icon_size, margin ;
637 if(!calculate_layouts(
638 hdc, checkmark_size, label_height, icon_size, margin)) {
639 return false ;
640 }
641
642 size.cx += margin + checkmark_size + margin + icon_size + margin ;
643
644 return true ;
645 }

651 LONG& checkmark_size,

775 bool create_tray(
776 const std::string& app_name,
777 const std::string& icon_path="",
778 unsigned char opacity=255,
779 bool round_corner=true) {
780 if(!util::string2wstring(app_name, app_name_)) {
781 return false ;
782 }
783
784 WNDCLASSW winc ;
785 winc.style = CS_HREDRAW | CS_VREDRAW ;
786 winc.lpfnWndProc = &FluentTray::callback ;
787 winc.cbClsExtra = 0 ;
788 winc.cbWndExtra = sizeof(LONG) * 2 ; // To store two-part address.
789 winc.hInstance = hinstance_ ;
790 winc.hIcon = LoadIcon(NULL, IDI_APPLICATION) ;
791 winc.hCursor = LoadCursor(NULL, IDC_ARROW) ;
792 winc.hbrBackground = GetSysColorBrush(COLOR_WINDOW) ;
793 winc.lpszMenuName = NULL ;
794 winc.lpszClassName = app_name_.c_str() ;
795
796 if(!RegisterClassW(&winc)) {
797 return false ;
798 }
799
800 hwnd_ = CreateWindowExW(
801 WS_EX_TOOLWINDOW | WS_EX_LAYERED,
802 app_name_.c_str(),
803 app_name_.c_str(),
804 WS_POPUPWINDOW,
805 0, 0, 100, 100,
806 NULL, NULL,
807 hinstance_, NULL
808 ) ;
809 if(!hwnd_) {
810 return false ;
811 }
812
813 // To access the this pointer inside the callback function,
814 // the address divide into the two part address.
815 LONG upper_addr, lower_addr ;
816 util::split_bits(this, upper_addr, lower_addr) ;
817
818 SetLastError(0) ;
819 if(!SetWindowLongW(hwnd_, 0, upper_addr) && GetLastError() != 0) {
820 return false ;
821 }
822 SetLastError(0) ;
823 if(!SetWindowLongW(hwnd_, sizeof(LONG), lower_addr) && GetLastError() != 0) {
824 return false ;
825 }
826
827 if(!SetLayeredWindowAttributes(hwnd_, 0, opacity, LWA_ALPHA)) {
828 return false ;
829 }
830
831 // Set rounded window for Windows 11 only.
832 if(round_corner) {
833#if defined(_MSC_VER) && _MSC_VER >= 1500
834 using RtlGetVersionType = NTSTATUS (WINAPI*)(PRTL_OSVERSIONINFOW) ;
835 const auto hmodule = LoadLibraryW(L"ntdll.dll") ;
836 if(!hmodule) {
837 return false ;
838 }
839
840 // cast to void* once to avoid warnings about type conversion.
841 const auto RtlGetVersion = reinterpret_cast<RtlGetVersionType>(
842 reinterpret_cast<void*>(GetProcAddress(hmodule, "RtlGetVersion"))) ;
843 if(!RtlGetVersion) {
844 FreeLibrary(hmodule) ;
845 return false ;
846 }
847
848 OSVERSIONINFOW vinfo ;
849 vinfo.dwOSVersionInfoSize = sizeof(decltype(vinfo)) ;
850 auto result = RtlGetVersion(&vinfo) ;
851 if(!FreeLibrary(hmodule)) {
852 return false ;
853 }
854
855 if(result != STATUS_SUCCESS) {
856 return false ;
857 }
858
859 // Check if the OS is Windows11
860 if(vinfo.dwBuildNumber >= 22000) {
861 auto pref = DWMWCP_ROUND ;
862 if(DwmSetWindowAttribute(
863 hwnd_,
864 DWMWA_WINDOW_CORNER_PREFERENCE,
865 &pref, sizeof(pref)) != S_OK) {
866 return false ;
867 }
868 }
869#endif // defined(_MSC_VER) && _MSC_VER >= 1500
870 }
871
872 if(!change_icon(icon_path)) {
873 return false ;
874 }
875
876 if(!set_font()) {
877 return false ;
878 }
879
880 status_ = TrayStatus::RUNNING ;
881
882 return true ;
883 }

928 bool update() {
929 if(status_ == TrayStatus::FAILED) {
930 return false ;
931 }
932
933 MSG msg ;
934 get_message(msg) ;
935
936 if(GetForegroundWindow() != hwnd_ && visible_) {
937 if(!hide_menu_window()) {
938 fail() ;
939 return false ;
940 }
941 }
942
943 for(std::size_t i = 0 ; i < menus_.size() ; i ++) {
944 auto& menu = menus_[i] ;
945 if(menu.is_mouse_over()) {
946 if(!mouse_is_over_[i]) {
947 if(!change_menu_back_color(menu, border_color_)) {
948 fail() ;
949 return false ;
950 }
951 }
952 mouse_is_over_[i] = true ;
953 }
954 else {
955 if(mouse_is_over_[i]) {
956 if(!change_menu_back_color(menu, back_color_)) {
957 fail() ;
958 return false ;
959 }
960 }
961 mouse_is_over_[i] = false ;
962 }
963 }
964

1004 // Initialize the color settings.
1005 if(back_color_ == CLR_INVALID) {
1006 // If the color is CLR_INVALID, it is determined from the theme.

1013 }
1014 }
1015 if(text_color_ == CLR_INVALID) {
1016 // The text color is automatically determined from the background color.
1017 text_color_ = calculate_text_color_(back_color_) ;
1018 if(text_color_ == CLR_INVALID) {
1019 return false ;
1020 }
1021 }

1031 for(auto& menu : menus_) {
1032 SIZE size ;
1033 if(!menu.calculate_required_dims(font_, size)) {
1034 return false ;

1427 if(auto self = get_instance()) {

1430 }
1431 }
